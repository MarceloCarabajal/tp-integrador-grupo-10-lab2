67 inline std::string trim(std::string str) {
68 size_t lSpaces = 0, rSpaces = 0, strEnd = 0;
69 std::string temp;
70 // contar espacios desde la izquierda
71 for (size_t i = 0; i < str.length(); i++) {
72 // si es una letra, salir del for
73 if (::isvalid::alphanumeric(str[i])) break;
74 if (str[i] == char(32)) lSpaces++;
75 }
76 // contar espacios desde la derecha
77 // se convierte length() a (int) porque i no puede ser de tipo size_t
78 // debido a que se usa decremento y al llegar a 0 y decrementar, un tipo
79 // size_t no puede ser negativo y vuelve a su valor maximo, generando un
80 // loop infinito
81 for (int i = (int)str.length(); i >= 0; i--) {
82 // si es una letra, salir del for
83 if (::isvalid::alphanumeric(str[i - 1])) break;
84 if (str[i - 1] == char(32)) rSpaces++;
85 }
86 // Calcular tamanio total sin espacios L y R
87 strEnd = str.length() - rSpaces;
88 // copiar str desde lSpaces hasta strEnd
89 for (size_t i = lSpaces; i < strEnd; i++) {
90 temp += str[i];
91 }
92 // si no se detectaron espacios, devolver str
93 if (lSpaces + rSpaces == 0) return str;
94 // sino devolver str trimeado
95 return temp;
96 }

162 inline int getLongestLine(std::string str) {
163 int maxLength = 0, charCount = 0;
164 for (int i = 0; i < (int)str.length(); i++) {
